* INSTALL
- unix

  #+BEGIN_SRC shell
  git clone https://github.com/xhcoding/doom-private.git ~/.config/doom
  cd ~/.emacs.d/
  make all
  #+END_SRC

- windows
  Open git bash
  #+BEGIN_SRC shell
  git clone https://github.com/xhcoding/doom-private.git ~/.config/doom
  cd ~/.emacs.d/
  bin/doom install
  #+END_SRC
